# Environment Variables Example
# Copy this file to .env.local and fill in your actual values

# Resend API Key (for sending emails)
# Sign up at: https://resend.com (Free: 3,000 emails/month)
RESEND_API_KEY=re_your_resend_api_key_here

# Google Maps API Key (for address autocomplete)
# Get from: https://console.cloud.google.com/
# Enable Places API and restrict to your domains
NEXT_PUBLIC_GOOGLE_MAPS_API_KEY=your_google_maps_api_key_here

# Supabase (Required for authentication and saved calculations)
# Sign up at: https://supabase.com (Free tier: 500MB database, 50K MAU)
NEXT_PUBLIC_SUPABASE_URL=your_supabase_project_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_supabase_anon_key
SUPABASE_SERVICE_ROLE_KEY=your_supabase_service_role_key

# JWT Secret (for session tokens)
# Generate a secure random string (32+ characters)
JWT_SECRET=your-secure-random-secret-key-change-this-in-production
